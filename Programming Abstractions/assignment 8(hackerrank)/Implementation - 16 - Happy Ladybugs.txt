#include <bits/stdc++.h>

using namespace std;

bool checkHappines(string b,int index){
    if(index == 0) return b[index+1] == b[index];
    if(index == b.size()-1) return b[index-1] == b[index];
    return b[index-1] == b[index] || b[index+1] == b[index];
}

// Complete the happyLadybugs function below.
string happyLadybugs(string b) {
    int arr[27] = {};
    bool wrongOrder = false;
    for (int i = 0; i < b.length(); i++) {
        if (b[i] == '_')  arr[26]++;
        else arr[b[i] - 'A']++;
        if(!wrongOrder){
            wrongOrder = !checkHappines(b,i);
        }
    }
    if(wrongOrder && arr[26] == 0) return "NO";
    for (int i = 0; i < 26; i++) {
      if (arr[i] == 1)
        return "NO";
    }
    return "YES";

}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    int g;
    cin >> g;
