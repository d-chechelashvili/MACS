#include <bits/stdc++.h>

using namespace std;

// Complete the gameOfThrones function below.
string gameOfThrones(string s) {
    int oddNum=0;
    int arr[26] = {};
    for (int i = 0; i < s.length(); i++) arr[s[i] - 'a']++;
    for(int i = 0 ; i < 26; i++){
        if(arr[i] % 2 != 0){
            oddNum++;
            if(oddNum > 1) return "NO";
        }
    }
    return "YES";
}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    string s;
    getline(cin, s);

    string result = gameOfThrones(s);

    fout << result << "\n";

    fout.close();

    return 0;
}
